import matplotlib.pyplot as plt
import numpy as np

# Generate random numbers
random_numbers = np.random.randint(0, 100, 5)  # Generate 5 random integers between 0 and 100

# Create a sequence of indices for the data points
indices = np.arange(len(random_numbers))

# Create a figure and axis objects
fig, (scatter_ax, plot_ax) = plt.subplots(nrows=1, ncols=2, figsize=(12, 6))

# Scatter plot
scatter_ax.scatter(indices, random_numbers, color='blue', label='Data Points')
scatter_ax.set_title('Scatter Plot')
scatter_ax.set_xlabel('Index')
scatter_ax.set_ylabel('Value')
scatter_ax.legend()

# Plot
plot_ax.plot(indices, random_numbers, color='red', marker='o', linestyle='-', label='Data Points')
plot_ax.set_title('Line Plot')
plot_ax.set_xlabel('Index')
plot_ax.set_ylabel('Value')
plot_ax.legend()

plt.tight_layout()
plt.show()
